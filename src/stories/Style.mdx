## Pattern installation styles

Install `postcss-import` and `tailwindcss` in the project.

<Unstyled>
  <Copy language='bash'>
    {`
      npm install postcss-import tailwindcss
    `}
  </Copy>
</Unstyled>

Create the file `postcss.config.js` in the root of the project.

<Unstyled>
  <Copy language='yaml'>
    {`
      module.exports = {
        plugins: {
          'postcss-import': {},
          tailwindcss: {},
          autoprefixer: {},
        },
      }
    `}
  </Copy>
</Unstyled>

The colors are defined in the root of the project in the file `tailwind.config.js` in the `colors` property.

<Unstyled>
  <Copy language='yaml'>
    {`
        colors: {
          primary: {
            regular: 'var(--color-primary-regular)',
            light: 'var(--color-primary-light)',
            dark: 'var(--color-primary-dark)',
          },
          secondary: {
            regular: 'var(--color-secondary-regular)',
            light: 'var(--color-secondary-light)',
            dark: 'var(--color-secondary-dark)',
          },
          neutral: {
            black: 'var(--color-neutral-black)',
            white: 'var(--color-neutral-white)',
            light: 'var(--color-neutral-light)',
            lightest: 'var(--color-neutral-lightest)',
            dark: 'var(--color-neutral-dark)',
          },
          feedback: {
            error: 'var(--color-feedback-error)',
            warning: 'var(--color-feedback-warning)',
            success: 'var(--color-feedback-success)',
            info: 'var(--color-feedback-info)',
          },
          complementary: {
            highlight: 'var(--color-complementary-highlight)',
          },
        },
    `}
  </Copy>
</Unstyled>

Create global styles for the colors and config tailwind in the file `**/global.css`.

<Unstyled>
  <Copy language='yaml'>
    {`
      @import '@coqueirodigital/react-components/dist/styles.css';
      @import 'tailwindcss/base';
      @import 'tailwindcss/components';
      @import 'tailwindcss/utilities';
      
      @layer base {
        :root {
          --color-primary-regular: #0d61ba;
          --color-primary-light: ##439bed;
          --color-primary-40: #0d61ba40;
          --color-primary-80: #0d61ba80;
          --color-primary-dark: #0c2445;

          --color-secondary-regular: #335250;
          --color-secondary-light: #bad5d1;
          --color-secondary-dark: #121e1e;

          --color-neutral-black: #131313;
          --color-neutral-white: #f9f9f9;
          --color-neutral-light: #c4c4c4;
          --color-neutral-lightest: #0000000a;
          --color-neutral-dark: #121e1e;

          --color-feedback-error: #f44336;
          --color-feedback-warning: #ff9800;
          --color-feedback-success: #4caf;
          --color-feedback-info: #2196f3;

          --color-complementary-highlight: #43b4ff;
          --color-complementary-highlight-80: #43b4ff;
        }
      }
    `}
  </Copy>
</Unstyled>
